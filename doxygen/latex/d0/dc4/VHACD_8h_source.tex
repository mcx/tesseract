\DoxyCodeLine{41 \textcolor{comment}{//}}

\DoxyCodeLine{45 \textcolor{comment}{// now when computing convex hulls, it is no longer a blocking operation.  All callback messages are still returned}}

\DoxyCodeLine{48 \textcolor{comment}{// the last approximation operation is complete and will dispatch any pending messages.}}
