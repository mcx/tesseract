199inline COW::Ptr createFCLCollisionObject(const std::string& name,
200 const int& type_id,
201 const CollisionShapesConst& shapes,
202 const tesseract_common::VectorIsometry3d& shape_poses,
203 bool enabled)
204{
205 // dont add object that does not have geometry
206 if (shapes.empty() || shape_poses.empty() || (shapes.size() != shape_poses.size()))
207 {
208 CONSOLE_BRIDGE_logDebug("ignoring link %s", name.c_str());
209 return nullptr;
210 }
211
212 auto new_cow = std::make_shared<COW>(name, type_id, shapes, shape_poses);
213
214 new_cow->m_enabled = enabled;
215 CONSOLE_BRIDGE_logDebug("Created collision object for link %s", new_cow->getName().c_str());
216 return new_cow;
217}

226inline void updateCollisionObjectFilters(const std::vector<std::string>& active,
227 const COW::Ptr& cow,
228 const std::unique_ptr<fcl::BroadPhaseCollisionManagerd>& static_manager,
229 const std::unique_ptr<fcl::BroadPhaseCollisionManagerd>& dynamic_manager)
230{
231 // For descrete checks we can check static to kinematic and kinematic to
232 // kinematic
233 if (!isLinkActive(active, cow->getName()))
234 {
235 if (cow->m_collisionFilterGroup != CollisionFilterGroups::StaticFilter)
236 {
237 std::vector<CollisionObjectPtr>& objects = cow->getCollisionObjects();
238 // This link was dynamic but is now static
239 for (auto& co : objects)
240 dynamic_manager->unregisterObject(co.get());
241
242 for (auto& co : objects)
243 static_manager->registerObject(co.get());
244 }
245 cow->m_collisionFilterGroup = CollisionFilterGroups::StaticFilter;
246 }
247 else
248 {
249 if (cow->m_collisionFilterGroup != CollisionFilterGroups::KinematicFilter)
250 {
251 std::vector<CollisionObjectPtr>& objects = cow->getCollisionObjects();
252 // This link was static but is now dynamic
253 for (auto& co : objects)
254 static_manager->unregisterObject(co.get());
255
256 for (auto& co : objects)
257 dynamic_manager->registerObject(co.get());
258 }
259 cow->m_collisionFilterGroup = CollisionFilterGroups::KinematicFilter;
260 }
261
262 // If the group is StaticFilter then the Mask is KinematicFilter, then StaticFilter groups can only collide with
263 // KinematicFilter groups. If the group is KinematicFilter then the mask is StaticFilter and KinematicFilter meaning
264 // that KinematicFilter groups can collide with both StaticFilter and KinematicFilter groups.
265 if (cow->m_collisionFilterGroup == CollisionFilterGroups::StaticFilter)
266 {
267 cow->m_collisionFilterMask = CollisionFilterGroups::KinematicFilter;
268 }
269 else
270 {
271 cow->m_collisionFilterMask = CollisionFilterGroups::StaticFilter | CollisionFilterGroups::KinematicFilter;
272 }
273}
