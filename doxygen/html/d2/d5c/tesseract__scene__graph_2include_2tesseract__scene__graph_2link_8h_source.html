200class Link

205
206 Link(std::string name) : name_(std::move(name)) { this->clear(); }
207 Link() = default;
208 ~Link() = default;
209 // Links are non-copyable as their name must be unique
210 Link(const Link& other) = delete;
211 Link& operator=(const Link& other) = delete;
212
213 Link(Link&& other) = default;
214 Link& operator=(Link&& other) = default;
215
216 const std::string& getName() const { return name_; }
217
219 Inertial::Ptr inertial;
220
222 std::vector<Visual::Ptr> visual;
223
225 std::vector<Collision::Ptr> collision;
226
227 void clear()
228 {
229 this->inertial.reset();
230 this->collision.clear();
231 this->visual.clear();
232 }
233
234 bool operator==(const Link& rhs) const;
235 bool operator!=(const Link& rhs) const;
236
237#ifndef SWIG
242 Link clone() const { return clone(name_); }
243
245 Link clone(const std::string& name) const
246 {
247 Link ret(name);
248 if (this->inertial)
249 {
250 ret.inertial = std::make_shared<Inertial>(*(this->inertial));
251 }
252 for (const auto& c : this->collision)
253 {
254 ret.collision.push_back(std::make_shared<Collision>(*c));
255 }
256 for (const auto& v : this->visual)
257 {
258 ret.visual.push_back(std::make_shared<Visual>(*v));
259 }
260 return ret;
261 }
262#endif // SWIG
263
264private:
265 std::string name_;
266 friend class boost::serialization::access;
267 template <class Archive>
268 void serialize(Archive& ar, const unsigned int version); // NOLINT
269};
