44std::shared_ptr<ClassBase> ClassLoader::createSharedInstance(const std::string& symbol_name,
45 const std::string& library_name,
46 const std::string& library_directory)
47{
48 boost::system::error_code ec;
49 boost::dll::shared_library lib;
50 if (library_directory.empty())
51 {
52 boost::filesystem::path sl(library_name);
53 boost::dll::load_mode::type mode =
54 boost::dll::load_mode::append_decorations | boost::dll::load_mode::search_system_folders;
55 lib = boost::dll::shared_library(sl, ec, mode);
56 }
57 else
58 {
59 boost::filesystem::path sl = boost::filesystem::path(library_directory) / library_name;
60 lib = boost::dll::shared_library(sl, ec, boost::dll::load_mode::append_decorations);
61 }
62
63 // Check if it failed to find or load library
64 if (ec)
65 throw std::runtime_error("Failed to find or load library: " + decorate(library_name, library_directory) +
66 " with error: " + ec.message());
67
68 // Check if library has symbol
69 if (!lib.has(symbol_name))
70 throw std::runtime_error("Failed to find symbol '" + symbol_name +
71 "' in library: " + decorate(library_name, library_directory));
72
73#if BOOST_VERSION >= 107600
74 boost::shared_ptr<ClassBase> plugin = boost::dll::import_symbol<ClassBase>(lib, symbol_name);
75#else
76 boost::shared_ptr<ClassBase> plugin = boost::dll::import<ClassBase>(lib, symbol_name);
77#endif
78 return std::shared_ptr<ClassBase>(plugin.get(), [plugin](ClassBase*) mutable { plugin.reset(); });
79}

148 const std::string& library_directory,

172 return std::vector<std::string>();

174
175 // Class `library_info` can extract information from a library
176 boost::dll::library_info inf(lib.location());
177
178 // Getting section from library
179 std::vector<std::string> sections = inf.sections();
180
181 auto search_fn = [include_hidden](const std::string& section) {
182 if (section.empty())
183 return true;
184
185 if (include_hidden)
186 return false;
187
188 return (section.substr(0, 1) == ".");
189 };
190
191 sections.erase(std::remove_if(sections.begin(), sections.end(), search_fn), sections.end());
192 return sections;

195std::string ClassLoader::decorate(const std::string& library_name, const std::string& library_directory)
196{
197 boost::filesystem::path sl;
198 if (library_directory.empty())
199 sl = boost::filesystem::path(library_name);
200 else
201 sl = boost::filesystem::path(library_directory) / library_name;
202
203 boost::filesystem::path actual_path =
204 (std::strncmp(sl.filename().string().c_str(), "lib", 3) != 0 ?
205 boost::filesystem::path((sl.has_parent_path() ? sl.parent_path() / L"lib" : L"lib").native() +
206 sl.filename().native()) :
207 sl);
208
209 actual_path += boost::dll::shared_library::suffix();
210 return actual_path.string();
211}
