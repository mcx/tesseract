127inline static void numericalJacobian(Eigen::Ref<Eigen::MatrixXd> jacobian,
128 const Eigen::Isometry3d& change_base,
129 const StateSolver& state_solver,
130 const std::vector<std::string>& joint_names,
131 const Eigen::Ref<const Eigen::VectorXd>& joint_values,
132 const std::string& link_name,
133 const Eigen::Ref<const Eigen::Vector3d>& link_point)
134{
135 Eigen::VectorXd njvals;
136 double delta = 0.001;
137 tesseract_common::TransformMap poses;
138 if (joint_names.empty())
139 poses = state_solver.getState(joint_values).link_transforms;
140 else
141 poses = state_solver.getState(joint_names, joint_values).link_transforms;
142
143 Eigen::Isometry3d pose = poses[link_name];
144 pose = change_base * pose;
145
146 for (int i = 0; i < static_cast<int>(joint_values.size()); ++i)
147 {
148 njvals = joint_values;
149 njvals[i] += delta;
150 tesseract_common::TransformMap updated_poses;
151 if (joint_names.empty())
152 updated_poses = state_solver.getState(njvals).link_transforms;
153 else
154 updated_poses = state_solver.getState(joint_names, njvals).link_transforms;
155
156 Eigen::Isometry3d updated_pose = updated_poses[link_name];
157 updated_pose = change_base * updated_pose;
158
159 Eigen::Vector3d temp = pose * link_point;
160 Eigen::Vector3d temp2 = updated_pose * link_point;
161 jacobian(0, i) = (temp2.x() - temp.x()) / delta;
162 jacobian(1, i) = (temp2.y() - temp.y()) / delta;
163 jacobian(2, i) = (temp2.z() - temp.z()) / delta;
164
165 Eigen::AngleAxisd r12(pose.rotation().transpose() * updated_pose.rotation()); // rotation from p1 -> p2
166 double theta = r12.angle();
167 theta = copysign(fmod(fabs(theta), 2.0 * M_PI), theta);
168 if (theta < -M_PI)
169 theta = theta + 2. * M_PI;
170 if (theta > M_PI)
171 theta = theta - 2. * M_PI;
172 Eigen::VectorXd omega = (pose.rotation() * r12.axis() * theta) / delta;
173 jacobian(3, i) = omega(0);
174 jacobian(4, i) = omega(1);
175 jacobian(5, i) = omega(2);
176 }

189inline void runCompareJacobian(StateSolver& state_solver,
190 const std::vector<std::string>& joint_names,
191 const Eigen::VectorXd& jvals,
192 const std::string& link_name,
193 const Eigen::Vector3d& link_point,
194 const Eigen::Isometry3d& change_base)
195{
196 Eigen::MatrixXd jacobian, numerical_jacobian;
197 jacobian.resize(6, jvals.size());
198
199 tesseract_common::TransformMap poses;
200
201 // The numerical jacobian orders things base on the provided joint list
202 // The order needs to be calculated to compare
203 std::vector<std::string> solver_jn = state_solver.getActiveJointNames();
204 std::vector<long> order;
205 if (joint_names.empty())
206 {
207 for (int i = 0; i < static_cast<int>(solver_jn.size()); ++i)
208 order.push_back(i);
209
210 poses = state_solver.getState(jvals).link_transforms;
211 jacobian = state_solver.getJacobian(jvals, link_name);
212 }
213 else
214 {
215 for (const auto& joint_name : solver_jn)
216 order.push_back(
217 std::distance(joint_names.begin(), std::find(joint_names.begin(), joint_names.end(), joint_name)));
218
219 poses = state_solver.getState(joint_names, jvals).link_transforms;
220 jacobian = state_solver.getJacobian(joint_names, jvals, link_name);
221 }
222
223 tesseract_common::jacobianChangeBase(jacobian, change_base);
224 tesseract_common::jacobianChangeRefPoint(jacobian, (change_base * poses[link_name]).linear() * link_point);
225
226 numerical_jacobian.resize(6, jvals.size());
227 numericalJacobian(numerical_jacobian, change_base, state_solver, joint_names, jvals, link_name, link_point);
228
229 for (int i = 0; i < 6; ++i)
230 {
231 for (int j = 0; j < static_cast<int>(jvals.size()); ++j)
232 {
233 EXPECT_NEAR(numerical_jacobian(i, order[static_cast<std::size_t>(j)]), jacobian(i, j), 1e-3);
234 }
235 }
