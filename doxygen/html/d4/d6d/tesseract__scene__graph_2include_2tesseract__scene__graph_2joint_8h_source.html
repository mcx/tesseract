292class Joint
293{
294public:

300 Joint(std::string name) : name_(std::move(name)) { this->clear(); }
301 Joint() = default;
302 ~Joint() = default;
303 // Joints are non-copyable as their name must be unique
304 Joint(const Joint& other) = delete;
305 Joint& operator=(const Joint& other) = delete;
306
307 Joint(Joint&& other) = default;

309
310 const std::string& getName() const { return name_; }

323 Eigen::Vector3d axis;

328
331 std::string parent_link_name;
332

335
337 JointDynamics::Ptr dynamics;
338
340 JointLimits::Ptr limits;
341
343 JointSafety::Ptr safety;
344
346 JointCalibration::Ptr calibration;
347
349 JointMimic::Ptr mimic;
350
351 void clear()
352 {
353 this->axis = Eigen::Vector3d(1, 0, 0);
354 this->child_link_name.clear();
355 this->parent_link_name.clear();
356 this->parent_to_joint_origin_transform.setIdentity();
357 this->dynamics.reset();
358 this->limits.reset();
359 this->safety.reset();
360 this->calibration.reset();
361 this->mimic.reset();
362 this->type = JointType::UNKNOWN;
363 }
364
365#ifndef SWIG
370 Joint clone() const { return clone(name_); }
371
372 /* Create a clone of current joint, with a new name. Child link name and parent link name are unchanged.
373 * All underlying properties, such as dynamics, limits... are copied as well.*/
374 Joint clone(const std::string& name) const
375 {
376 Joint ret(name);
377 ret.axis = this->axis;
378 ret.child_link_name = this->child_link_name;
379 ret.parent_link_name = this->parent_link_name;
380 ret.parent_to_joint_origin_transform = this->parent_to_joint_origin_transform;
381 ret.type = this->type;
382 if (this->dynamics)
383 {
384 ret.dynamics = std::make_shared<JointDynamics>(*(this->dynamics));
385 }
386 if (this->limits)
387 {
388 ret.limits = std::make_shared<JointLimits>(*(this->limits));
389 }
390 if (this->safety)
391 {
392 ret.safety = std::make_shared<JointSafety>(*(this->safety));
393 }
394 if (this->calibration)
395 {
396 ret.calibration = std::make_shared<JointCalibration>(*(this->calibration));
397 }
398 if (this->mimic)
399 {
400 ret.mimic = std::make_shared<JointMimic>(*(this->mimic));
401 }
402 return ret;
403 }
404 bool operator==(const Joint& rhs) const;
405 bool operator!=(const Joint& rhs) const;
406#endif // SWIG
407
408private:
409 std::string name_;
410
411 friend class boost::serialization::access;
412 template <class Archive>
413 void serialize(Archive& ar, const unsigned int version); // NOLINT
414};
