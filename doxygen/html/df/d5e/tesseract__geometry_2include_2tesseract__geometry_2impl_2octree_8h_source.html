146 static bool isNodeCollapsible(octomap::OcTree& octree, octomap::OcTreeNode* node)
147 {
148 if (!octree.nodeChildExists(node, 0))
149 return false;
150
151 double occupancy_threshold = octree.getOccupancyThres();
152
153 const octomap::OcTreeNode* firstChild = octree.getNodeChild(node, 0);
154 if (octree.nodeHasChildren(firstChild) || firstChild->getOccupancy() < occupancy_threshold)
155 return false;
156
157 for (unsigned int i = 1; i < 8; i++)
158 {
159 // comparison via getChild so that casts of derived classes ensure
160 // that the right == operator gets called
161 if (!octree.nodeChildExists(node, i))
162 return false;
163
164 if (octree.nodeHasChildren(octree.getNodeChild(node, i)))
165 return false;
166
167 if (octree.getNodeChild(node, i)->getOccupancy() < occupancy_threshold)
168 return false;
169 }
170
171 return true;
172 }

174 static bool pruneNode(octomap::OcTree& octree, octomap::OcTreeNode* node)
175 {
176 if (!isNodeCollapsible(octree, node))
177 return false;
178
179 // set value to children's values (all assumed equal)
180 node->copyData(*(octree.getNodeChild(node, 0)));
181
182 // delete children (known to be leafs at this point!)
183 for (unsigned int i = 0; i < 8; i++)
184 {
185 octree.deleteNodeChild(node, i);
186 }
187
188 return true;
189 }

191 // NOLINTNEXTLINE(misc-no-recursion)
192 static void pruneRecurs(octomap::OcTree& octree,
193 octomap::OcTreeNode* node,
194 unsigned int depth,
195 unsigned int max_depth,
196 unsigned int& num_pruned)
197 {
198 assert(node);
199
200 if (depth < max_depth)
201 {
202 for (unsigned int i = 0; i < 8; i++)
203 {
204 if (octree.nodeChildExists(node, i))
205 {
206 pruneRecurs(octree, octree.getNodeChild(node, i), depth + 1, max_depth, num_pruned);
207 }
208 }
209 } // end if depth
210
211 else
212 {
213 // max level reached
214 if (pruneNode(octree, node))
215 {
216 num_pruned++;
217 }
218 }
