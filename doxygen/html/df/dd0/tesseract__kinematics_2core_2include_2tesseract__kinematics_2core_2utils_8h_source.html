55inline static void numericalJacobian(Eigen::Ref<Eigen::MatrixXd> jacobian,
56 const Eigen::Isometry3d& change_base,
57 const tesseract_kinematics::ForwardKinematics& kin,
58 const Eigen::Ref<const Eigen::VectorXd>& joint_values,
59 const std::string& link_name,
60 const Eigen::Ref<const Eigen::Vector3d>& link_point)
61{
62 Eigen::VectorXd njvals;
63 double delta = 1e-8;
64 tesseract_common::TransformMap poses = kin.calcFwdKin(joint_values);
65 Eigen::Isometry3d pose = poses[link_name];
66 pose = change_base * pose;
67
68 for (int i = 0; i < static_cast<int>(joint_values.size()); ++i)
69 {
70 njvals = joint_values;
71 njvals[i] += delta;
72 Eigen::Isometry3d updated_pose = kin.calcFwdKin(njvals)[link_name];
73 updated_pose = change_base * updated_pose;
74
75 Eigen::Vector3d temp = pose * link_point;
76 Eigen::Vector3d temp2 = updated_pose * link_point;
77 jacobian(0, i) = (temp2.x() - temp.x()) / delta;
78 jacobian(1, i) = (temp2.y() - temp.y()) / delta;
79 jacobian(2, i) = (temp2.z() - temp.z()) / delta;
80
81 Eigen::VectorXd omega = (pose.rotation() * tesseract_common::calcRotationalError(pose.rotation().transpose() *
82 updated_pose.rotation())) /
83 delta;
84 jacobian(3, i) = omega(0);
85 jacobian(4, i) = omega(1);
86 jacobian(5, i) = omega(2);
87 }
88}

139inline static bool solvePInv(const Eigen::Ref<const Eigen::MatrixXd>& A,
140 const Eigen::Ref<const Eigen::VectorXd>& b,
141 Eigen::Ref<Eigen::VectorXd> x)
142{
143 const double eps = 0.00001; // TODO: Turn into class member var
144 const double lambda = 0.01; // TODO: Turn into class member var
145
146 if ((A.rows() == 0) || (A.cols() == 0))
147 {
148 CONSOLE_BRIDGE_logError("Empty matrices not supported in solvePinv()");
149 return false;
150 }
151
152 if (A.rows() != b.size())
153 {
154 CONSOLE_BRIDGE_logError("Matrix size mismatch: A(%d, %d), b(%d)", A.rows(), A.cols(), b.size());
155 return false;
156 }
157
158 // Calculate A+ (pseudoinverse of A) = V S+ U*, where U* is Hermition of U (just transpose if all values of U are
159 // real)
160 // in order to solve Ax=b -> x*=A+ b
161 Eigen::JacobiSVD<Eigen::MatrixXd> svd(A, Eigen::ComputeThinU | Eigen::ComputeThinV);
162 const Eigen::MatrixXd& U = svd.matrixU();
163 const Eigen::VectorXd& Sv = svd.singularValues();
164 const Eigen::MatrixXd& V = svd.matrixV();
165
166 // calculate the reciprocal of Singular-Values
167 // damp inverse with lambda so that inverse doesn't oscillate near solution
168 long int nSv = Sv.size();
169 Eigen::VectorXd inv_Sv(nSv);
170 for (long int i = 0; i < nSv; ++i)
171 {
172 if (fabs(Sv(i)) > eps)
173 inv_Sv(i) = 1 / Sv(i);
174 else
175 inv_Sv(i) = Sv(i) / (Sv(i) * Sv(i) + lambda * lambda);
176 }
177 x = V * inv_Sv.asDiagonal() * U.transpose() * b;
178 return true;
179}

190inline static bool dampedPInv(const Eigen::Ref<const Eigen::MatrixXd>& A,
191 Eigen::Ref<Eigen::MatrixXd> P,
192 const double eps = 0.011,
193 const double lambda = 0.01)
194{
195 if ((A.rows() == 0) || (A.cols() == 0))
196 {
197 CONSOLE_BRIDGE_logError("Empty matrices not supported in dampedPInv()");
198 return false;
199 }
200
201 // Calculate A+ (pseudoinverse of A) = V S+ U*, where U* is Hermition of U (just transpose if all values of U are
202 // real)
203 // in order to solve Ax=b -> x*=A+ b
204 Eigen::JacobiSVD<Eigen::MatrixXd> svd(A, Eigen::ComputeThinU | Eigen::ComputeThinV);
205 const Eigen::MatrixXd& U = svd.matrixU();
206 const Eigen::VectorXd& Sv = svd.singularValues();
207 const Eigen::MatrixXd& V = svd.matrixV();
208
209 // calculate the reciprocal of Singular-Values
210 // damp inverse with lambda so that inverse doesn't oscillate near solution
211 long int nSv = Sv.size();
212 Eigen::VectorXd inv_Sv(nSv);
213 for (long int i = 0; i < nSv; ++i)
214 {
215 if (fabs(Sv(i)) > eps)
216 inv_Sv(i) = 1 / Sv(i);
217 else
218 {
219 inv_Sv(i) = Sv(i) / (Sv(i) * Sv(i) + lambda * lambda);
220 }
221 }
222 P = V * inv_Sv.asDiagonal() * U.transpose();
223 return true;
224}

234inline bool isNearSingularity(const Eigen::Ref<const Eigen::MatrixXd>& jacobian, double threshold = 0.01)
235{
236 Eigen::JacobiSVD<Eigen::MatrixXd> svd(jacobian, Eigen::ComputeThinU | Eigen::ComputeThinV);
237 const Eigen::VectorXd& sv = svd.singularValues();
238 return (sv.tail(1).value() < threshold);

302inline Manipulability calcManipulability(const Eigen::Ref<const Eigen::MatrixXd>& jacobian)
303{
304 Manipulability manip;
305 Eigen::MatrixXd jacob_linear = jacobian.topRows(3);
306 Eigen::MatrixXd jacob_angular = jacobian.bottomRows(3);
307
308 auto fn = [](const Eigen::MatrixXd& m) {
309 ManipulabilityEllipsoid data;
310 Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> sm(m, Eigen::DecompositionOptions::EigenvaluesOnly);
311 data.eigen_values = sm.eigenvalues().real();
312
313 // Set eigenvalues near zero to zero. This also implies zero volume
314 for (Eigen::Index i = 0; i < data.eigen_values.size(); ++i)
315 {
316 if (tesseract_common::almostEqualRelativeAndAbs(data.eigen_values[i], 0))
317 data.eigen_values[i] = +0;
318 }
319
320 // If the minimum eigen value is approximately zero set measure and condition to max double
321 if (tesseract_common::almostEqualRelativeAndAbs(data.eigen_values.minCoeff(), 0))
322 {
323 data.measure = std::numeric_limits<double>::max();
324 data.condition = std::numeric_limits<double>::max();
325 }
326 else
327 {
328 data.condition = data.eigen_values.maxCoeff() / data.eigen_values.minCoeff();
329 data.measure = std::sqrt(data.condition);
330 }
331
332 data.volume = std::sqrt(data.eigen_values.prod());
333
334 return data;
335 };
336
337 Eigen::MatrixXd a = jacobian * jacobian.transpose();
338 Eigen::MatrixXd a_linear = jacob_linear * jacob_linear.transpose();
339 Eigen::MatrixXd a_angular = jacob_angular * jacob_angular.transpose();
340 manip.m = fn(a);
341 manip.m_linear = fn(a_linear);
342 manip.m_angular = fn(a_angular);
343
344 Eigen::MatrixXd a_inv = a.inverse();
345 Eigen::MatrixXd a_linear_inv = a_linear.inverse();
346 Eigen::MatrixXd a_angular_inv = a_angular.inverse();
347 manip.f = fn(a_inv);
348 manip.f_linear = fn(a_linear_inv);
349 manip.f_angular = fn(a_angular_inv);
350
351 return manip;
352}

358template <typename FloatType>
359inline void getRedundantSolutionsHelper(std::vector<VectorX<FloatType>>& redundant_sols,
360 const Eigen::Ref<const Eigen::VectorXd>& sol,
361 const Eigen::MatrixX2d& limits,
362 std::vector<Eigen::Index>::const_iterator current_index,
363 std::vector<Eigen::Index>::const_iterator end_index)
364{
365 double val{ 0 };
366 for (; current_index != end_index; ++current_index)
367 {
368 if (std::isinf(limits(*current_index, 0)))
369 {
370 std::stringstream ss;
371 ss << "Lower limit of joint " << *current_index << " is infinite; no redundant solutions will be generated"
372 << std::endl;
373 CONSOLE_BRIDGE_logWarn(ss.str().c_str());
374 }
375 else
376 {
377 val = sol[*current_index];
378 while ((val -= (2.0 * M_PI)) > limits(*current_index, 0) ||
379 tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 0)))
380 {
381 // It not guaranteed that the provided solution is within limits so this check is needed
382 if (val < limits(*current_index, 1) ||
383 tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 1)))
384 {
385 Eigen::VectorXd new_sol = sol;
386 new_sol[*current_index] = val;
387
388 if (tesseract_common::satisfiesPositionLimits<double>(new_sol, limits))
389 {
390 tesseract_common::enforcePositionLimits<double>(new_sol, limits);
391 redundant_sols.push_back(new_sol.template cast<FloatType>());
392 }
393
394 getRedundantSolutionsHelper<FloatType>(redundant_sols, new_sol, limits, current_index + 1, end_index);
395 }
396 }
397 }
398
399 if (std::isinf(limits(*current_index, 1)))
400 {
401 std::stringstream ss;
402 ss << "Upper limit of joint " << *current_index << " is infinite; no redundant solutions will be generated"
403 << std::endl;
404 CONSOLE_BRIDGE_logWarn(ss.str().c_str());
405 }
406 else
407 {
408 val = sol[*current_index];
409 while ((val += (2.0 * M_PI)) < limits(*current_index, 1) ||
410 tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 1)))
411 {
412 // It not guaranteed that the provided solution is within limits so this check is needed
413 if (val > limits(*current_index, 0) ||
414 tesseract_common::almostEqualRelativeAndAbs(val, limits(*current_index, 0)))
415 {
416 Eigen::VectorXd new_sol = sol;
417 new_sol[*current_index] = val;
418
419 if (tesseract_common::satisfiesPositionLimits<double>(new_sol, limits))
420 {
421 tesseract_common::enforcePositionLimits<double>(new_sol, limits);
422 redundant_sols.push_back(new_sol.template cast<FloatType>());
423 }
424
425 getRedundantSolutionsHelper<FloatType>(redundant_sols, new_sol, limits, current_index + 1, end_index);
426 }
427 }
428 }
429 }
430}

439template <typename FloatType>
440inline std::vector<VectorX<FloatType>> getRedundantSolutions(const Eigen::Ref<const VectorX<FloatType>>& sol,
441 const Eigen::MatrixX2d& limits,
442 const std::vector<Eigen::Index>& redundancy_capable_joints)
443{
444 if (redundancy_capable_joints.empty())
445 return {};
446
447 for (const Eigen::Index& idx : redundancy_capable_joints)
448 {
449 if (idx >= sol.size())
450 {
451 std::stringstream ss;
452 ss << "Redundant joint index " << idx << " is greater than or equal to the joint state size (" << sol.size()
453 << ")";
454 throw std::runtime_error(ss.str());
455 }
456 }
457
458 std::vector<VectorX<FloatType>> redundant_sols;
459 getRedundantSolutionsHelper<FloatType>(redundant_sols,
460 sol.template cast<double>(),
461 limits,
462 redundancy_capable_joints.begin(),
463 redundancy_capable_joints.end());
464 return redundant_sols;
465}

493inline void harmonizeTowardZero(Eigen::Ref<VectorX<FloatType>> qs)
494{
495 const static auto pi = FloatType(M_PI);
496 const static auto two_pi = FloatType(2.0 * M_PI);
497
498 for (Eigen::Index i = 0; i < qs.rows(); ++i)
499 {
500 FloatType diff = std::fmod(qs[i] + pi, two_pi);
501 qs[i] = (diff < 0) ? (diff + pi) : (diff - pi);
502 }
