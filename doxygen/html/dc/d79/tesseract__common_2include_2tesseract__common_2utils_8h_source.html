57template <typename... Args>
58std::string strFormat(const std::string& format, Args... args)
59{
60 int size_s = std::snprintf(nullptr, 0, format.c_str(), args...) + 1; // Extra space for '\0'
61 if (size_s <= 0)
62 throw std::runtime_error("Error during formatting.");
63
64 auto size = static_cast<size_t>(size_s);
65 auto buf = std::make_unique<char[]>(size); // NOLINT
66 std::snprintf(buf.get(), size, format.c_str(), args...);
67 return std::string(buf.get(), buf.get() + size - 1); // We don't want the '\0' inside
68}

209template <typename T>
210bool isIdentical(
211 const std::vector<T>& vec1,
212 const std::vector<T>& vec2,
213 bool ordered = true,
214 const std::function<bool(const T&, const T&)>& equal_pred = [](const T& v1, const T& v2) { return v1 == v2; },
215 const std::function<bool(const T&, const T&)>& comp = [](const T& v1, const T& v2) { return v1 < v2; })
216{
217 if (vec1.size() != vec2.size())
218 return false;
219
220 if (ordered)
221 return std::equal(vec1.begin(), vec1.end(), vec2.begin(), equal_pred);
222
223 std::vector<T> v1 = vec1;
224 std::vector<T> v2 = vec2;
225 std::sort(v1.begin(), v1.end(), comp);
226 std::sort(v2.begin(), v2.end(), comp);
227 return std::equal(v1.begin(), v1.end(), v2.begin(), equal_pred);
228}

236template <typename KeyValueContainerType, typename ValueType>
237bool isIdenticalMap(
238 const KeyValueContainerType& map_1,
239 const KeyValueContainerType& map_2,
240 const std::function<bool(const ValueType&, const ValueType&)>& value_eq =
241 [](const ValueType& v1, const ValueType& v2) { return v1 == v2; })
242{
243 if (map_1.size() != map_2.size())
244 return false;
245
246 for (const auto& entry : map_1)
247 {
248 // Check if the key exists
249 const auto cp = map_2.find(entry.first);
250 if (cp == map_2.end())
251 return false;
252 // Check if the value is the same
253 if (!value_eq(cp->second, entry.second))
254 return false;
255 }
256 return true;
257}

477template <typename FloatType>
478bool toNumeric(const std::string& s, FloatType& value)
479{
480 if (s.empty())
481 return false;
482
483 std::stringstream ss;
484 ss.imbue(std::locale::classic());
485
486 ss << s;
487
488 FloatType out;
489 ss >> out;
490
491 if (ss.fail() || !ss.eof())
492 return false;
493
494 value = out;
495 return true;
496}
