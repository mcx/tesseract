611 auto num_in_edges = static_cast<int>(boost::in_degree(vertex, graph));

615 tree_ = false;

708 std::vector<std::string> getLinkChildrenHelper(Vertex start_vertex) const
709 {
710 const auto& graph = static_cast<const Graph&>(*this);
711 std::vector<std::string> child_link_names;
712
713 std::map<Vertex, size_t> index_map;
714 boost::associative_property_map<std::map<Vertex, size_t>> prop_index_map(index_map);
715
716 std::map<Vertex, boost::default_color_type> color_map;
717 boost::associative_property_map<std::map<Vertex, boost::default_color_type>> prop_color_map(color_map);
718
719 int c = 0;
720 Graph::vertex_iterator i, iend;
721 for (boost::tie(i, iend) = boost::vertices(graph); i != iend; ++i, ++c)
722 boost::put(prop_index_map, *i, c);
723
724 children_detector vis(child_link_names);
725 // NOLINTNEXTLINE
726 boost::breadth_first_search(
727 graph,
728 start_vertex,
729 boost::visitor(vis).root_vertex(start_vertex).vertex_index_map(prop_index_map).color_map(prop_color_map));
730
731 return child_link_names;
732 }
