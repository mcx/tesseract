605 void discover_vertex(u vertex, const g& graph)
606 {
607 auto num_in_edges = static_cast<int>(boost::in_degree(vertex, graph));
608
609 if (num_in_edges > 1)
610 {
611 tree_ = false;

613 }

704 std::vector<std::string> getLinkChildrenHelper(Vertex start_vertex) const
705 {
706 const auto& graph = static_cast<const Graph&>(*this);
707 std::vector<std::string> child_link_names;
708
709 std::map<Vertex, size_t> index_map;
710 boost::associative_property_map<std::map<Vertex, size_t>> prop_index_map(index_map);
711
712 std::map<Vertex, boost::default_color_type> color_map;
713 boost::associative_property_map<std::map<Vertex, boost::default_color_type>> prop_color_map(color_map);
714
715 int c = 0;
716 Graph::vertex_iterator i, iend;
717 for (boost::tie(i, iend) = boost::vertices(graph); i != iend; ++i, ++c)
718 boost::put(prop_index_map, *i, c);
719
720 children_detector vis(child_link_names);
721 // NOLINTNEXTLINE
722 boost::breadth_first_search(
723 graph,
724 start_vertex,
725 boost::visitor(vis).root_vertex(start_vertex).vertex_index_map(prop_index_map).color_map(prop_color_map));
726
727 return child_link_names;
728 }
