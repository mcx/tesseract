307GetAverageSupport(const btConvexShape* shape, const btVector3& localNormal, btScalar& outsupport, btVector3& outpt)
308{
309 btVector3 ptSum(0, 0, 0);
310 btScalar ptCount = 0;
311 btScalar maxSupport = -1000;
312
313 const auto* pshape = dynamic_cast<const btPolyhedralConvexShape*>(shape);
314 if (pshape != nullptr)
315 {
316 int nPts = pshape->getNumVertices();
317
318 for (int i = 0; i < nPts; ++i)
319 {
320 btVector3 pt;
321 pshape->getVertex(i, pt);
322
323 btScalar sup = pt.dot(localNormal);
324 if (sup > maxSupport + BULLET_EPSILON)
325 {
326 ptCount = 1;
327 ptSum = pt;
328 maxSupport = sup;
329 }
330 else if (sup < maxSupport - BULLET_EPSILON)
331 {
332 }
333 else
334 {
335 ptCount += 1;
336 ptSum += pt;
337 }
338 }
339 outsupport = maxSupport;
340 outpt = ptSum / ptCount;
341 }
342 else
343 {
344 // The margins are set to zero for most shapes, but for a sphere the margin is used so must use
345 // localGetSupportingVertex instead of localGetSupportingVertexWithoutMargin.
346 outpt = shape->localGetSupportingVertex(localNormal);
347 outsupport = localNormal.dot(outpt);

450inline void calculateContinuousData(ContactResult* col,
451 const btCollisionObjectWrapper* cow,
452 const btVector3& pt_world,
453 const btVector3& normal_world,
454 const btTransform& link_tf_inv,
455 size_t link_index)
456{
457 assert(dynamic_cast<const CastHullShape*>(cow->getCollisionShape()) != nullptr);
458 const auto* shape = static_cast<const CastHullShape*>(cow->getCollisionShape());
459 assert(shape != nullptr);
460
461 // Get the start and final location of the shape
462 btTransform shape_tfWorld0 = cow->getWorldTransform();
463 btTransform shape_tfWorld1 = cow->getWorldTransform() * shape->m_t01;
464
465 // Given the shapes final location calculate the links transform at the final location
466 Eigen::Isometry3d s = col->transform[link_index].inverse() * convertBtToEigen(shape_tfWorld0);
467 col->cc_transform[link_index] = convertBtToEigen(shape_tfWorld1) * s.inverse();
468
469 // Get the normal in the local shapes coordinate system at start and final location
470 btVector3 shape_normalLocal0 = normal_world * shape_tfWorld0.getBasis();
471 btVector3 shape_normalLocal1 = normal_world * shape_tfWorld1.getBasis();
472
473 // Calculate the contact point at the start location using the casted normal vector in thapes local coordinate system
474 btVector3 shape_ptLocal0;
475 btScalar shape_localsup0{ std::numeric_limits<btScalar>::max() };
476 GetAverageSupport(shape->m_shape, shape_normalLocal0, shape_localsup0, shape_ptLocal0);
477 btVector3 shape_ptWorld0 = shape_tfWorld0 * shape_ptLocal0;
478
479 // Calculate the contact point at the final location using the casted normal vector in thapes local coordinate system
480 btVector3 shape_ptLocal1;
481 btScalar shape_localsup1{ std::numeric_limits<btScalar>::max() };
482 GetAverageSupport(shape->m_shape, shape_normalLocal1, shape_localsup1, shape_ptLocal1);
483 btVector3 shape_ptWorld1 = shape_tfWorld1 * shape_ptLocal1;
484
485 btScalar shape_sup0 = normal_world.dot(shape_ptWorld0);
486 btScalar shape_sup1 = normal_world.dot(shape_ptWorld1);
487
488 // TODO: this section is potentially problematic. think hard about the math
489 if (shape_sup0 - shape_sup1 > BULLET_SUPPORT_FUNC_TOLERANCE)
490 {
491 // LCOV_EXCL_START
492 col->cc_time[link_index] = 0;
493 col->cc_type[link_index] = ContinuousCollisionType::CCType_Time0;
494 // LCOV_EXCL_STOP
495 }
496 else if (shape_sup1 - shape_sup0 > BULLET_SUPPORT_FUNC_TOLERANCE)
497 {
498 // LCOV_EXCL_START
499 col->cc_time[link_index] = 1;
500 col->cc_type[link_index] = ContinuousCollisionType::CCType_Time1;
501 // LCOV_EXCL_STOP
502 }
503 else
504 {
505 // Given the contact point at the start and final location along with the casted contact point
506 // the time between 0 and 1 can be calculated along the path between the start and final location contact occurs.
507 btScalar l0c = (pt_world - shape_ptWorld0).length();
508 btScalar l1c = (pt_world - shape_ptWorld1).length();
509
510 col->nearest_points_local[link_index] =
511 convertBtToEigen(link_tf_inv * (shape_tfWorld0 * ((shape_ptLocal0 + shape_ptLocal1) / 2.0)));
512 col->cc_type[link_index] = ContinuousCollisionType::CCType_Between;
513
514 if (l0c + l1c < BULLET_LENGTH_TOLERANCE)
515 {
516 col->cc_time[link_index] = .5; // LCOV_EXCL_LINE
517 }
518 else
519 {
520 col->cc_time[link_index] = static_cast<double>(l0c / (l0c + l1c));
521 }
522 }
523}

1013inline COW::Ptr makeCastCollisionObject(const COW::Ptr& cow)
1014{
1015 COW::Ptr new_cow = cow->clone();
1016
1017 btTransform tf;
1018 tf.setIdentity();
1019
1020 if (btBroadphaseProxy::isConvex(new_cow->getCollisionShape()->getShapeType()))
1021 {
1022 assert(dynamic_cast<btConvexShape*>(new_cow->getCollisionShape()) != nullptr);
1023 auto* convex = static_cast<btConvexShape*>(new_cow->getCollisionShape()); // NOLINT
1024 assert(convex->getShapeType() != CUSTOM_CONVEX_SHAPE_TYPE); // This checks if the collision object is already a
1025 // cast collision object
1026
1027 auto shape = std::make_shared<CastHullShape>(convex, tf);
1028 assert(shape != nullptr);
1029
1030 new_cow->manage(shape);
1031 new_cow->setCollisionShape(shape.get());
1032 }
1033 else if (btBroadphaseProxy::isCompound(new_cow->getCollisionShape()->getShapeType()))
1034 {
1035 assert(dynamic_cast<btCompoundShape*>(new_cow->getCollisionShape()) != nullptr);
1036 auto* compound = static_cast<btCompoundShape*>(new_cow->getCollisionShape()); // NOLINT
1037 auto new_compound =
1038 std::make_shared<btCompoundShape>(BULLET_COMPOUND_USE_DYNAMIC_AABB, compound->getNumChildShapes());
1039
1040 for (int i = 0; i < compound->getNumChildShapes(); ++i)
1041 {
1042 if (btBroadphaseProxy::isConvex(compound->getChildShape(i)->getShapeType()))
1043 {
1044 auto* convex = static_cast<btConvexShape*>(compound->getChildShape(i)); // NOLINT
1045 assert(convex->getShapeType() != CUSTOM_CONVEX_SHAPE_TYPE); // This checks if already a cast collision object
1046
1047 btTransform geomTrans = compound->getChildTransform(i);
1048
1049 auto subshape = std::make_shared<CastHullShape>(convex, tf);
1050 assert(subshape != nullptr);
1051
1052 new_cow->manage(subshape);
1053 subshape->setMargin(BULLET_MARGIN);
1054 new_compound->addChildShape(geomTrans, subshape.get());
1055 }
1056 else if (btBroadphaseProxy::isCompound(compound->getChildShape(i)->getShapeType()))
1057 {
1058 auto* second_compound = static_cast<btCompoundShape*>(compound->getChildShape(i)); // NOLINT
1059 auto new_second_compound =
1060 std::make_shared<btCompoundShape>(BULLET_COMPOUND_USE_DYNAMIC_AABB, second_compound->getNumChildShapes());
1061 for (int j = 0; j < second_compound->getNumChildShapes(); ++j)
1062 {
1063 assert(!btBroadphaseProxy::isCompound(second_compound->getChildShape(j)->getShapeType()));
1064 assert(dynamic_cast<btConvexShape*>(second_compound->getChildShape(j)) != nullptr);
1065
1066 auto* convex = static_cast<btConvexShape*>(second_compound->getChildShape(j)); // NOLINT
1067 assert(convex->getShapeType() != CUSTOM_CONVEX_SHAPE_TYPE); // This checks if already a cast collision object
1068
1069 btTransform geomTrans = second_compound->getChildTransform(j);
1070
1071 auto subshape = std::make_shared<CastHullShape>(convex, tf);
1072 assert(subshape != nullptr);
1073
1074 new_cow->manage(subshape);
1075 subshape->setMargin(BULLET_MARGIN);
1076 new_second_compound->addChildShape(geomTrans, subshape.get());
1077 }
1078
1079 btTransform geomTrans = compound->getChildTransform(i);
1080
1081 new_cow->manage(new_second_compound);
1082 new_second_compound->setMargin(BULLET_MARGIN); // margin: compound. seems to
1083 // have no effect when positive
1084 // but has an effect when
1085 // negative
1086
1087 new_compound->addChildShape(geomTrans, new_second_compound.get());
1088 }
1089 else
1090 {
1091 throw std::runtime_error("I can only collision check convex shapes and compound shapes made of convex shapes");
1092 }
1093 }
1094
1095 new_compound->setMargin(BULLET_MARGIN); // margin: compound. seems to
1096 // have no effect when positive
1097 // but has an effect when
1098 // negative
1099 new_cow->manage(new_compound);
1100 new_cow->setCollisionShape(new_compound.get());
1101 new_cow->setWorldTransform(cow->getWorldTransform());
1102 }

1104 {
1105 throw std::runtime_error("I can only collision check convex shapes and compound shapes made of convex shapes");
1106 }
