309GetAverageSupport(const btConvexShape* shape, const btVector3& localNormal, btScalar& outsupport, btVector3& outpt)
310{
311 btVector3 ptSum(0, 0, 0);
312 btScalar ptCount = 0;
313 btScalar maxSupport = -1000;
314
315 const auto* pshape = dynamic_cast<const btPolyhedralConvexShape*>(shape);
316 if (pshape != nullptr)
317 {
318 int nPts = pshape->getNumVertices();
319
320 for (int i = 0; i < nPts; ++i)
321 {
322 btVector3 pt;
323 pshape->getVertex(i, pt);
324
325 btScalar sup = pt.dot(localNormal);
326 if (sup > maxSupport + BULLET_EPSILON)
327 {
328 ptCount = 1;
329 ptSum = pt;
330 maxSupport = sup;
331 }
332 else if (sup < maxSupport - BULLET_EPSILON)
333 {
334 }
335 else
336 {
337 ptCount += 1;
338 ptSum += pt;
339 }
340 }
341 outsupport = maxSupport;
342 outpt = ptSum / ptCount;
343 }
344 else
345 {
346 // The margins are set to zero for most shapes, but for a sphere the margin is used so must use
347 // localGetSupportingVertex instead of localGetSupportingVertexWithoutMargin.
348 outpt = shape->localGetSupportingVertex(localNormal);
349 outsupport = localNormal.dot(outpt);

452inline void calculateContinuousData(ContactResult* col,
453 const btCollisionObjectWrapper* cow,
454 const btVector3& pt_world,
455 const btVector3& normal_world,
456 const btTransform& link_tf_inv,
457 size_t link_index)
458{
459 assert(dynamic_cast<const CastHullShape*>(cow->getCollisionShape()) != nullptr);
460 const auto* shape = static_cast<const CastHullShape*>(cow->getCollisionShape());
461 assert(shape != nullptr);
462
463 // Get the start and final location of the shape
464 btTransform shape_tfWorld0 = cow->getWorldTransform();
465 btTransform shape_tfWorld1 = cow->getWorldTransform() * shape->m_t01;
466
467 // Given the shapes final location calculate the links transform at the final location
468 Eigen::Isometry3d s = col->transform[link_index].inverse() * convertBtToEigen(shape_tfWorld0);
469 col->cc_transform[link_index] = convertBtToEigen(shape_tfWorld1) * s.inverse();
470
471 // Get the normal in the local shapes coordinate system at start and final location
472 btVector3 shape_normalLocal0 = normal_world * shape_tfWorld0.getBasis();
473 btVector3 shape_normalLocal1 = normal_world * shape_tfWorld1.getBasis();
474
475 // Calculate the contact point at the start location using the casted normal vector in thapes local coordinate system
476 btVector3 shape_ptLocal0;
477 btScalar shape_localsup0{ std::numeric_limits<btScalar>::max() };
478 GetAverageSupport(shape->m_shape, shape_normalLocal0, shape_localsup0, shape_ptLocal0);
479 btVector3 shape_ptWorld0 = shape_tfWorld0 * shape_ptLocal0;
480
481 // Calculate the contact point at the final location using the casted normal vector in thapes local coordinate system
482 btVector3 shape_ptLocal1;
483 btScalar shape_localsup1{ std::numeric_limits<btScalar>::max() };
484 GetAverageSupport(shape->m_shape, shape_normalLocal1, shape_localsup1, shape_ptLocal1);
485 btVector3 shape_ptWorld1 = shape_tfWorld1 * shape_ptLocal1;
486
487 btScalar shape_sup0 = normal_world.dot(shape_ptWorld0);
488 btScalar shape_sup1 = normal_world.dot(shape_ptWorld1);
489
490 // TODO: this section is potentially problematic. think hard about the math
491 if (shape_sup0 - shape_sup1 > BULLET_SUPPORT_FUNC_TOLERANCE)
492 {
493 // LCOV_EXCL_START
494 col->cc_time[link_index] = 0;
495 col->cc_type[link_index] = ContinuousCollisionType::CCType_Time0;
496 // LCOV_EXCL_STOP
497 }
498 else if (shape_sup1 - shape_sup0 > BULLET_SUPPORT_FUNC_TOLERANCE)
499 {
500 // LCOV_EXCL_START
501 col->cc_time[link_index] = 1;
502 col->cc_type[link_index] = ContinuousCollisionType::CCType_Time1;
503 // LCOV_EXCL_STOP
504 }
505 else
506 {
507 // Given the contact point at the start and final location along with the casted contact point
508 // the time between 0 and 1 can be calculated along the path between the start and final location contact occurs.
509 btScalar l0c = (pt_world - shape_ptWorld0).length();
510 btScalar l1c = (pt_world - shape_ptWorld1).length();
511
512 col->nearest_points_local[link_index] =
513 convertBtToEigen(link_tf_inv * (shape_tfWorld0 * ((shape_ptLocal0 + shape_ptLocal1) / 2.0)));
514 col->cc_type[link_index] = ContinuousCollisionType::CCType_Between;
515
516 if (l0c + l1c < BULLET_LENGTH_TOLERANCE)
517 {
518 col->cc_time[link_index] = .5; // LCOV_EXCL_LINE
519 }
520 else
521 {
522 col->cc_time[link_index] = static_cast<double>(l0c / (l0c + l1c));
523 }
524 }
525}

1015inline COW::Ptr makeCastCollisionObject(const COW::Ptr& cow)
1016{
1017 COW::Ptr new_cow = cow->clone();
1018
1019 btTransform tf;
1020 tf.setIdentity();
1021
1022 if (btBroadphaseProxy::isConvex(new_cow->getCollisionShape()->getShapeType()))
1023 {
1024 assert(dynamic_cast<btConvexShape*>(new_cow->getCollisionShape()) != nullptr);
1025 auto* convex = static_cast<btConvexShape*>(new_cow->getCollisionShape()); // NOLINT
1026 assert(convex->getShapeType() != CUSTOM_CONVEX_SHAPE_TYPE); // This checks if the collision object is already a
1027 // cast collision object
1028
1029 auto shape = std::make_shared<CastHullShape>(convex, tf);
1030 assert(shape != nullptr);
1031
1032 new_cow->manage(shape);
1033 new_cow->setCollisionShape(shape.get());
1034 }
1035 else if (btBroadphaseProxy::isCompound(new_cow->getCollisionShape()->getShapeType()))
1036 {
1037 assert(dynamic_cast<btCompoundShape*>(new_cow->getCollisionShape()) != nullptr);
1038 auto* compound = static_cast<btCompoundShape*>(new_cow->getCollisionShape()); // NOLINT
1039 auto new_compound =
1040 std::make_shared<btCompoundShape>(BULLET_COMPOUND_USE_DYNAMIC_AABB, compound->getNumChildShapes());
1041
1042 for (int i = 0; i < compound->getNumChildShapes(); ++i)
1043 {
1044 if (btBroadphaseProxy::isConvex(compound->getChildShape(i)->getShapeType()))
1045 {
1046 auto* convex = static_cast<btConvexShape*>(compound->getChildShape(i)); // NOLINT
1047 assert(convex->getShapeType() != CUSTOM_CONVEX_SHAPE_TYPE); // This checks if already a cast collision object
1048
1049 btTransform geomTrans = compound->getChildTransform(i);
1050
1051 auto subshape = std::make_shared<CastHullShape>(convex, tf);
1052 assert(subshape != nullptr);
1053
1054 new_cow->manage(subshape);
1055 subshape->setMargin(BULLET_MARGIN);
1056 new_compound->addChildShape(geomTrans, subshape.get());
1057 }
1058 else if (btBroadphaseProxy::isCompound(compound->getChildShape(i)->getShapeType()))
1059 {
1060 auto* second_compound = static_cast<btCompoundShape*>(compound->getChildShape(i)); // NOLINT
1061 auto new_second_compound =
1062 std::make_shared<btCompoundShape>(BULLET_COMPOUND_USE_DYNAMIC_AABB, second_compound->getNumChildShapes());
1063 for (int j = 0; j < second_compound->getNumChildShapes(); ++j)
1064 {
1065 assert(!btBroadphaseProxy::isCompound(second_compound->getChildShape(j)->getShapeType()));
1066 assert(dynamic_cast<btConvexShape*>(second_compound->getChildShape(j)) != nullptr);
1067
1068 auto* convex = static_cast<btConvexShape*>(second_compound->getChildShape(j)); // NOLINT
1069 assert(convex->getShapeType() != CUSTOM_CONVEX_SHAPE_TYPE); // This checks if already a cast collision object
1070
1071 btTransform geomTrans = second_compound->getChildTransform(j);
1072
1073 auto subshape = std::make_shared<CastHullShape>(convex, tf);
1074 assert(subshape != nullptr);
1075
1076 new_cow->manage(subshape);
1077 subshape->setMargin(BULLET_MARGIN);
1078 new_second_compound->addChildShape(geomTrans, subshape.get());
1079 }
1080
1081 btTransform geomTrans = compound->getChildTransform(i);
1082
1083 new_cow->manage(new_second_compound);
1084 new_second_compound->setMargin(BULLET_MARGIN); // margin: compound. seems to
1085 // have no effect when positive
1086 // but has an effect when
1087 // negative
1088
1089 new_compound->addChildShape(geomTrans, new_second_compound.get());
1090 }
1091 else
1092 {
1093 throw std::runtime_error("I can only collision check convex shapes and compound shapes made of convex shapes");
1094 }
1095 }
1096
1097 new_compound->setMargin(BULLET_MARGIN); // margin: compound. seems to
1098 // have no effect when positive
1099 // but has an effect when
1100 // negative
1101 new_cow->manage(new_compound);
1102 new_cow->setCollisionShape(new_compound.get());
1103 new_cow->setWorldTransform(cow->getWorldTransform());
1104 }
1105 else
1106 {

1108 }
