89 static std::string toArchiveStringXML(const SerializableType& archive_type, const std::string& name = "")
90 {
91 std::stringstream ss;
92 { // Must be scoped because all data is not written until the oost::archive::xml_oarchive goes out of scope
93 boost::archive::xml_oarchive oa(ss);
94
95 // Boost uses the same function for serialization and deserialization so it requires a non-const reference
96 // Because we are only serializing here it is safe to cast away const
97 if (name.empty())
98 oa << boost::serialization::make_nvp<SerializableType>("archive_type",
99 const_cast<SerializableType&>(archive_type)); // NOLINT
100 else
101 oa << boost::serialization::make_nvp<SerializableType>(name.c_str(),
102 const_cast<SerializableType&>(archive_type)); // NOLINT
103 }
104
105 return ss.str();
106 }
